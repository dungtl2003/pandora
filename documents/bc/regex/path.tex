}

    PATH\_IN\_EXPRESSION -> PATH\_EXPR\_SEGMENT

    \hspace{5.1cm}(\w{::} PATH\_EXPR\_SEGMENT)*

    PATH\_EXPR\_SEGMENT -> PATH\_IDENT\_SEGMENT

    \hspace{5.1cm}(\w{::} GENERIC\_ARGS)?

    PATH\_IDENT\_SEGMENT -> IDENTIFIER | \w{super} | \w{self}

    GENERIC\_ARGS ->

    \hspace{3.05cm}\w{<}(GENERIC\_ARG\w{,})* GENERIC\_ARG\w{,}?\w{>}

    GENERIC\_ARG -> TYPE
    \end{rg}
}
