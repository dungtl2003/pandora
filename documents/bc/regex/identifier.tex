\newcommand{\regexidentifier}{%
    \begin{rg}
    IDENTIFIER -> NON\_KEYWORD\_IDENTIFIER

    \hspace{3.5cm}| RAW\_IDENTIFIER

    NON\_KEYWORD\_IDENTIFIER -> IDENTIFIER\_OR\_KEYWORD

    \hspace*{0pt}\hfill{\small \textit{(without keyword)}}

    RAW\_IDENTIFIER -> \w{r}\w{\#}IDENTIFIER\_OR\_KEYWORD

    IDENTIFIER\_OR\_KEYWORD -> ID\_START ID\_CONTINUE*

    ID\_START -> \w{\_} | XID\_Start | EMOJI\_SYMBOL

    ID\_CONTINUE -> XID\_Continue | EMOJI\_SYMBOL
    \end{rg}
}
